		ROOT = {
			independent = yes
			NOT = { same_realm = FROM }
		}

		if = {
			limit = {
				ROOT = { religion = triadic }
				FROM = { olrox_is_good_religion_trigger = yes }
			}
			ROOT = { piety = -50 }
		}

		if = {
			limit = { defender = { is_offmap_governor = shou_lung } }
			attacker = {
				sound_effect = china_angered_emperor
				detract_grace_super_huge_effect = yes
			}
		}

			}

				ROOT = {

					is_rival = FROM
				}
			}

		}

						}

						10 = { give_nickname = nick_the_victorious }
						10 = { give_nickname = nick_the_bold }
						10 = { give_nickname = nick_the_conqueror }
					}
				}

				if = {
					limit = {
						has_nickname = nick_the_bastard
					}
					give_nickname = nick_the_conqueror
				}

				if = {
					limit = {
						has_dlc = "Legacy of Rome"
						OR = {
							has_landed_title = e_cormanthor
							has_landed_title = e_the_heartlands

			limit = {

					is_rival = ROOT
				}
			}
			FROM = { piety = 40 }
		}
		ROOT = {
			show_scope_change = no
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 8.0
			}
		}

			show_scope_change = no
			prestige = 200
			participation_scaled_prestige = 200

				limit = {
					has_dlc = "Legacy of Rome"
					OR = {
						has_landed_title = e_cormanthor
