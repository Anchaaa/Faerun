rd_age_up_dwarf = {
	if = {
		limit = {
			age = 40
		}
		break = yes
	}
	else = {
		set_variable = {
			which = final_age
			value = 40
		}
		export_to_variable = {
			which = starting_age
			value = age
		}
		subtract_variable = {
			which = final_age
			which = starting_age
		}
		add_age = final_age
	}
}
