is_known_undead = {
	custom_tooltip = {
		text = IS_KNOWN_UNDEAD_TOOLTIP

		is_undead = yes
		trigger_if = {
			limit = {
				OR = {
					trait = vampire
					trait = vampire_spawn
				}
			}

			has_character_modifier = known_vamp_modifier
		}
	}
}
is_secretly_undead = {
	trigger_if = {
		limit = {
			OR = {
				trait = vampire
				trait = vampire_spawn
			}
		}

		NOT = { has_character_modifier = known_vamp_modifier }
	}
	trigger_else_if = {
		limit = { has_character_flag = secret_lich }

		is_undead = no
	}
	trigger_else = {
		always = no
	}
}
