male_patrician_trigger = {
	trigger_if = {
		limit = {
			NOT = {
				has_game_rule = {
					name = gender
					value = all
				}
			}
		}
		is_female = no
		NOR = {
			primary_title = {
				OR = {
					has_law = enatic_succession
					has_law = enatic_cognatic_succession
				}
			}
			liege = {
				primary_title = {
					OR = {
						has_law = enatic_succession
						has_law = enatic_cognatic_succession
					}
				}
			}
		}
	}
}
