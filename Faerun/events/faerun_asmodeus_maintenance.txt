character_event = {
	id = asmodeusmaint.3
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			AND = {
				mother = { 
					   OR = { 
						trait = infernal_beauty 
						trait = infernal_wit 
						trait = infernal_strength 
						trait = infernal_contractee 
						}
					}
				dynasty = mother
				}
			AND = {
				true_father = { 
					   OR = { 
						trait = infernal_beauty 
						trait = infernal_wit 
						trait = infernal_strength 
						trait = infernal_contractee 
						}
					}
				dynasty = true_father
				}
		}
		NOT = { trait = bastard }
	}

	immediate = { 
		if = {
			limit = {
				mother = { trait = infernal_beauty }
				dynasty = mother
			}
			add_trait = infernal_beauty
			add_trait = beautiful
                	add_trait = planetouched
			break = yes
		}
		else_if = {
			limit = {
				mother = { trait = infernal_wit }
				dynasty = mother
			}
			add_trait = infernal_wit
			add_trait = genius
                	add_trait = planetouched
			break = yes
		}
		else_if = {
			limit = {
				mother = { trait = infernal_strength }
				dynasty = mother
			}
			add_trait = infernal_strength
			add_trait = strong
                	add_trait = planetouched
			break = yes
		}		
		else_if = {
			limit = {
				mother = { trait = infernal_contractee }
				dynasty = mother
			}
			add_trait = infernal_contractee
                	add_trait = planetouched
			break = yes
		}
		else_if = {
			limit = {
				true_father = { trait = infernal_beauty  }
				dynasty = true_father
			}
			add_trait = infernal_beauty 
			add_trait = beautiful
                	add_trait = planetouched
			break = yes
		}
		else_if = {
			limit = {
				true_father = { trait = infernal_wit }
				dynasty = true_father
			}
			add_trait = infernal_wit
			add_trait = genius
                	add_trait = planetouched
			break = yes
		}
		else_if = {
			limit = {
				true_father = { trait = infernal_strength }
				dynasty = true_father
			}
			add_trait = infernal_strength
			add_trait = strong
                	add_trait = planetouched
			break = yes
		}
		else_if = {
			limit = {
				true_father = { trait = infernal_contractee }
				dynasty = true_father
			}
			add_trait = infernal_contractee
                	add_trait = planetouched
			break = yes
		}
	}
}
