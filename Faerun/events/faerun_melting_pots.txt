character_event = {
	id = frmeltpot.1
	desc = FRMELTPOT.1
	picture = GFX_evt_emissary
	
	is_triggered_only = yes
	
	trigger = {
		higher_tier_than = DUKE
		is_tribal = no
		OR = {
			culture = calishite
			culture = chondathan
			culture = arnadar
			culture = tashalan
			culture = lapal
			culture = tethyrian
			culture = shoon
			culture = borderer
		}
		capital_scope = {
			region = border_kingdoms_region
		}
		NOT = { has_global_flag = borderer_melting_pot }
		any_realm_province = {
			region = border_kingdoms_region
			OR = {
				culture = calishite
				culture = shoon
			}
		}
		any_realm_province = {
			region = border_kingdoms_region
			culture = chondathan
		}
		any_realm_province = {
			region = border_kingdoms_region
			OR = {
				culture = arnadar
				culture = tashalan
				culture = lapal
				culture = tethyrian
			}
		}
	}
	
	option = {
		name = FRMELTPOT.1a
		set_global_flag = borderer_melting_pot
		set_global_flag = ruler_became_borderer
		random_realm_province = {
			limit = {
				region = border_kingdoms_region
				OR = {
					culture = calishite
					culture = chondathan
					culture = arnadar
					culture = tashalan
					culture = lapal
					culture = tethyrian
					culture = shoon
				}
			}
			culture = borderer
		}
		prestige = -50
		culture = borderer
	}
	
	option = {
		name = FRMELTPOT.1b
		set_global_flag = borderer_melting_pot
		set_character_flag = refused_borderer_flip
		random_realm_province = {
			limit = {
				region = border_kingdoms_region
				OR = {
					culture = calishite
					culture = chondathan
					culture = arnadar
					culture = tashalan
					culture = lapal
					culture = tethyrian
					culture = shoon
				}
			}
			culture = borderer
			set_province_flag = flipped_to_borderer
		}
	}
}

character_event = {
	id = frmeltpot.2
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		any_realm_province = {
			region = border_kingdoms_region
			has_global_flag = borderer_melting_pot
			NOT = { has_province_flag = flipped_to_borderer }
			OR = {
				culture = calishite
				culture = chondathan
				culture = arnadar
				culture = tashalan
				culture = lapal
				culture = tethyrian
				culture = shoon
			}
		}
	}
	
	weight_multiplier = {
		days = 1
		modifier = {
			factor = 3
			culture = borderer
		}
		modifier = {
			factor = 2
			has_global_flag = ruler_became_borderer
		}
	}
	
	immediate = {
		random_realm_province = {
			limit = {
				region = border_kingdoms_region
				NOT = { has_province_flag = flipped_to_borderer }
				OR = {
					culture = calishite
					culture = chondathan
					culture = arnadar
					culture = tashalan
					culture = lapal
					culture = tethyrian
					culture = shoon
				}
			}
			province_event = { id = frmeltpot.3 }
		}
	}
}
