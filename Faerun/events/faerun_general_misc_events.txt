character_event = { #lower fertility every child for true immortal characters
	id = generalmisc.3
	hide_window = yes
	is_triggered_only = yes #on_birth
	desc = evilcatlocbugevent

	trigger = {
		OR = {
			father = {
				faerun_immortal_race_trigger = yes
			}
			mother = {
				faerun_immortal_race_trigger = yes
			}
		}
	}

	immediate = {
		mother = {
			if = {
				limit = {
					faerun_immortal_race_trigger = yes
				}
				random = {
					chance = 50
					fertility = -0.1
				}
			}
		}
		father = {
			if = {
				limit = {
					faerun_immortal_race_trigger = yes
				}
				random = {
					chance = 50
					fertility = -0.1
				}
			}
		}
	}
}
