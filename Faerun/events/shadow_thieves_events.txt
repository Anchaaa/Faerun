# Has either Schemer Impaler or Seduction already - Replace the existing trait
character_event = {
	id = STH.1901
	picture = GFX_evt_shady_meeting
	border = GFX_event_normal_frame_intrigue

	is_triggered_only = yes

	desc = {
		trigger = { trait = schemer }
		picture = GFX_evt_shady_meeting
		text = EVTDESCA_STH_1901
	}

	desc = {
		trigger = { trait = impaler }
		picture = GFX_evt_torture
		text = EVTDESCB_STH_1901
	}

	desc = {
		trigger = {
			OR = {
				trait = seducer
				trait = seductress
			}
		}
		picture = GFX_evt_lovers
		text = EVTDESCC_STH_1901
	}

	option = { # Change to Schemer
		name = EVTOPTA_STH_1901

		trigger = {
			NOT = { trait = schemer }
		}

		if = {
			limit = { trait = impaler }
			remove_trait = impaler
		}
		if = {
			limit = { trait = seducer }
			remove_trait = seducer
		}
		if = {
			limit = { trait = seductress }
			remove_trait = seductress
		}
		add_trait = schemer
		clr_character_flag = choosing_shadow_art
	}

	option = { # Change to Impaler
		name = EVTOPTB_STH_1901

		trigger = {
			NOT = { trait = impaler }
		}

		if = {
			limit = { trait = schemer }
			remove_trait = schemer
		}
		if = {
			limit = { trait = seducer }
			remove_trait = seducer
		}
		if = {
			limit = { trait = seductress }
			remove_trait = seductress
		}
		add_trait = impaler
		clr_character_flag = choosing_shadow_art
	}

	option = { # Change to Seduction
		name = EVTOPTC_STH_1901

		trigger = {
			NOR = {
				trait = seducer
				trait = seductress
			}
		}

		if = {
			limit = { trait = schemer }
			remove_trait = schemer
		}
		if = {
			limit = { trait = impaler }
			remove_trait = impaler
		}
		if = {
			limit = { is_female = no }
			add_trait = seducer
		}
		if = {
			limit = { is_female = yes }
			add_trait = seductress
		}
		clr_character_flag = choosing_shadow_art
	}

	option = { # Change your mind
		name = EVTOPTD_STH_1901

		add_society_currency_minor_effect = yes
		clr_character_flag = choosing_shadow_art
	}
}

# Become Schemer
character_event = {
	id = STH.1902
	desc = EVTDESC_STH_1902
	picture = GFX_evt_shady_meeting
	border = GFX_event_normal_frame_intrigue

	is_triggered_only = yes

	option = {
		name = EVTOPTA_STH_1902

		if = {
			limit = {
				NOT = { trait = deceitful }
				NOT = { trait = honest }
			}
			add_trait = deceitful
		}
		if = {
			limit = {
				trait = honest
			}
			remove_trait = honest
		}
		prestige = 25
		add_character_modifier = {
				modifier = street_smarts
				duration = 3650
			}
		character_event = { id = STH.1903 days = 60 random = 30 }

		ai_chance = {
			factor = 1
		}
	}

	option = {
		name = EVTOPTB_STH_1902

		random_list = {
			50 = { change_intrigue = 1 }
			40 = {
				modifier = {
					factor = 0
					trait = patient
				}
				add_trait = patient
			}
			10 = {
				modifier = {
					factor = 0
					OR = {
						trait = groomed
						trait = uncouth
					}
				}
				add_trait = groomed
			}
		}
		character_event = { id = STH.1903 days = 60 random = 30 }

		ai_chance = {
			factor = 1
		}
	}
}

# Hostile Guild
character_event = {
	id = STH.403
	desc = EVTDESC_STH_403
	picture = GFX_evt_bandits

	trigger = {
		NOT = { society_member_of = shadow_thieves }
		higher_tier_than = BARON
		any_demesne_province = {
	    has_province_modifier = shadow_thieves_guild
		}
	}

	mean_time_to_happen = {
		months = 60
	}

	option = { #lose money
		name = EVTOPTA_STH_403
		ai_chance = {
			factor = 30
		}
		scaled_wealth = -0.05
	}

	option = { #destroy guild
		name = EVTOPTB_STH_403
		ai_chance = {
			factor = 30
		}
		scaled_wealth = -0.2
		random_list = {
			80 = {
				random_demesne_province = {
					limit = {
						has_province_modifier = shadow_thieves_guild
					}
					remove_province_modifier = shadow_thieves_guild
				}
			}
			20 = {
				prestige = -10
			}
		}
	}

	option = { #join
		name = EVTOPTC_STH_403
		trigger = {
			is_in_society = no
			age = 16
			intrigue = 10
			NOT = {
				trait = honest
				culture = monster
			}
			OR = {
				z_rogue_classes = yes
				AND = {
					trait = patron_mask
					OR = {
						z_cleric = yes
						z_warlock = yes
					}
				}
			}
		}
		ai_chance = {
			factor = 10
			modifier = {
				factor = 3
				trait = deceitful
			}
			modifier = {
				factor = 3
				trait = schemer
			}
			modifier = {
				factor = 3
				trait = ambitious
			}
			modifier = {
				factor = 3
				trait = greedy
			}
			modifier = {
				factor = 3
				trait = envious
			}
		}
		join_society = shadow_thieves
	}
}
