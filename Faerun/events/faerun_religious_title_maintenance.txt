namespace = faithcleanup

character_event = { # 
	id = faithcleanup.1
	desc = FAITHCLEANUP_1
	picture = GFX_evt_stressed_ruler

	is_triggered_only = yes

	trigger = {
		has_landed_title = d_cult_of_shajar
		NAND = {
			religion = nogari
			OR = {
				trait = patron_shajar
				has_patron_god = no
			}
		}
	}
	option = {
		name = EVTOPTA_FAITHCLEANUP_1
		unsafe_destroy_landed_title = d_cult_of_shajar
	}
}
