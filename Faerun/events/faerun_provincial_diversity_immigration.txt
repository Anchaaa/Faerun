namespace = PDI

character_event = {
  id = PDI.1
  desc = EVTDESCPDI1
  picture = GFX_evt_culture_change
  border = GFX_event_normal_frame_economy

  only_playable = yes

  trigger = {
    any_realm_province = {
      religion = ROOT
      has_gnome_pops = yes
      NOT = { has_province_modifier = recent_gnome_emigration }
    }
    any_realm_province = {
      religion = ROOT
      has_gnome_pops = no
    }
  }

  mean_time_to_happen = {
    years = 10
  }

  immediate = {
    random_realm_province = {
      limit = {
        religion = ROOT
        has_gnome_pops = yes
        NOT = { has_province_modifier = recent_gnome_emigration }
      }
      if = {
        limit = { has_province_modifier = gnome_big }
        add_province_modifier = {
          name = recent_gnome_emigration
          hidden = yes
          years = 7
        }
      }
      else = {
        add_province_modifier = {
          name = recent_gnome_emigration
          hidden = yes
          years = 20
        }
      }
      save_event_target_as = origin_province
    }
    random_realm_province = {
      limit = {
        religion = ROOT
        has_gnome_pops = no
      }
      add_province_modifier = {
        name = gnome_small
        hidden = yes
        duration = -1
      }
      save_event_target_as = destination_province
    }
  }

  option = {
    name = EVTOPTPDI
  }
}

    any_realm_province = {
      religion = ROOT
      has_verbeeg_pops = yes
      NOT = { has_province_modifier = recent_verbeeg_emigration }
    }
    any_realm_province = {
      religion = ROOT
      has_verbeeg_pops = no
    }
  }

  mean_time_to_happen = {
    years = 10
  }

  immediate = {
    random_realm_province = {
      limit = {
        religion = ROOT
        has_verbeeg_pops = yes
        NOT = { has_province_modifier = recent_verbeeg_emigration }
      }
      if = {
        limit = { has_province_modifier = verbeeg_big }
        add_province_modifier = {
          name = recent_verbeeg_emigration
          hidden = yes
          years = 7
        }
      }
      else = {
        add_province_modifier = {
          name = recent_verbeeg_emigration
          hidden = yes
          years = 20
        }
      }
      save_event_target_as = origin_province
    }
    random_realm_province = {
      limit = {
        religion = ROOT
        has_verbeeg_pops = no
      }
      add_province_modifier = {
        name = verbeeg_small
        hidden = yes
        duration = -1
      }
      save_event_target_as = destination_province
    }
  }

  option = {
    name = EVTOPTPDI
  }
}
