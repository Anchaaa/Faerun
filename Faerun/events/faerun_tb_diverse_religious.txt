character_event = {
	id = divrel.1
	desc = DIVREL.1
	picture = GFX_evt_bishop
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	option = {
		name = DIVREL.1a
		piety = 50
		if = {
			limit = {
				trait = stressed
			}
			random = {
				chance = 35
				remove_trait = stressed
			}
		}
	}
	
	option = {
		name = DIVREL.1b
		any_realm_character = {
			limit = {
				trait = patron_helm
			}
			opinion = {
				modifier = honored_helm
				who = ROOT
				years = 1
			}
		}
	}
	
	option = {
		name = DIVREL.1c
		add_character_modifier = {
			name = helped_helm_clergy
			years = 1
		}
		if = {
			limit = {
				trait = cynical
			}
			random = {
				chance = 50
				remove_trait = cynical
			}
		}
		else_if = {
			limit = {
				NOT = { trait = zealous }
			}
			random = {
				chance = 15
				add_trait = zealous
			}
		}
	}
}
