# Hostile Guild
character_event = {
	id = TELF.403
	desc = EVTDESC_TELF_403
	picture = GFX_evt_bandits

	trigger = {
		NOT = { society_member_of = shadow_telflamm }
		higher_tier_than = BARON
		any_demesne_province = {
	    has_province_modifier = shadow_telflamm_guild
		}
	}

	mean_time_to_happen = {
		months = 60
	}

	option = { #lose money
		name = EVTOPTA_TELF_403
		ai_chance = {
			factor = 30
		}
		scaled_wealth = -0.05
	}

	option = { #destroy guild
		name = EVTOPTB_TELF_403
		ai_chance = {
			factor = 30
		}
		scaled_wealth = -0.2
		random_list = {
			80 = {
				random_demesne_province = {
					limit = {
						has_province_modifier = shadow_telflamm_guild
					}
					remove_province_modifier = shadow_telflamm_guild
				}
			}
			20 = {
				prestige = -10
			}
		}
	}

	option = { #join
		name = EVTOPTC_TELF_403
		trigger = {
			is_in_society = no
			age = 16
			intrigue = 10
			NOT = {
				trait = honest
				culture = monster
			}
			OR = {
				z_rogue_classes = yes
				AND = {
					trait = patron_mask
					OR = {
						z_cleric = yes
						z_warlock = yes
					}
				}
			}
		}
		ai_chance = {
			factor = 10
			modifier = {
				factor = 3
				trait = deceitful
			}
			modifier = {
				factor = 3
				trait = schemer
			}
			modifier = {
				factor = 3
				trait = ambitious
			}
			modifier = {
				factor = 3
				trait = greedy
			}
			modifier = {
				factor = 3
				trait = envious
			}
		}
		join_society = shadow_telflamm
	}
}

#Mission Trigger: Heist
character_event = {
	id = TELF.1430
	hide_window = yes

	trigger = {
		society_member_of = shadow_telflamm
		higher_tier_than = BARON
		any_demesne_province = {
			has_province_modifier = shadow_telflamm_guild
		}
	}

	mean_time_to_happen = {
		months = 400
	}

	immediate = {
		random_independent_ruler = {
			limit = {
				NOR = {
					has_non_aggression_pact_with = ROOT
					society_member_of = shadow_telflamm
				}
				primary_title = {
					is_landless_type_title = no
				}
				controls_religion = no
				holy_order = no
				mercenary = no
				is_landed = yes
				is_within_diplo_range = ROOT
			}
			save_event_target_as = target_ruler
		}
		if = {
			limit = {
				event_target:target_ruler = { always = yes }
				is_within_diplo_range = event_target:target_ruler
			}
			random_society_member = {
				limit = { society_rank = { rank = 4 society = shadow_telflamm } }
				character_event = { id = STH.1431 }
			}
		}
	}
}

#From on_character_ask_to_join_society
character_event = {
    id = TELF.100
    hide_window = yes
    is_triggered_only = yes
    trigger = {
    	FROM = { leader = { society_member_of = shadow_telflamm } }
    }
    immediate = {
    	if = { #Find a non-prisoner member
    		limit = { FROM = { leader = { prisoner = yes } } }
    		FROM = {
    			any_society_member = {
    				limit = {
    					society_member_of = shadow_telflamm
    					society_rank == 4
    					prisoner = no
    				}
    				character_event = { id = TELF.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = shadow_telflamm
    					society_rank == 3
    					prisoner = no
    				}
    				character_event = { id = TELF.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = shadow_telflamm
    					society_rank == 2
    					prisoner = no
    				}
    				character_event = { id = TELF.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = shadow_telflamm
    					society_rank == 1
    					prisoner = no
    				}
    				character_event = { id = TELF.101 }
    				break = yes
    			}
    		}
			#Everyone is somehow in prison so just use teh leader anyway
    	}
    	FROM = { leader = { character_event = { id = TELF.101 } } }
    }
}

########################AI JOIN AND AI RANK UP###########################

character_event = { #force the AI to join a society
	id = TELF.110
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = { #if he's already a member, will force a rank up 25% of the time
				ai = yes
				prisoner = no
				NOT = { trait = incapable }
				society_member_of = shadow_telflamm
				society_rank <= 3
				society_can_rank_up = yes
			}
			random_list = {
				75 = {}
				25 = {
					society_rank_up = 1
				}
			}
		}
		if = { #force the joining event on AI
			limit = {
				ai = yes
				prisoner = no
				NOT = { trait = incapable }
				is_adult = yes
				is_in_society = no
				controls_religion = no
				block_general_event_trigger = no
			}
			character_event = { id = TELF.111 }
		}
	}
}
