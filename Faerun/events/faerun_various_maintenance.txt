	}

	immediate = {
		any_demesne_province = {
			limit = {

				limit = {
					has_province_modifier = prosperity_modifier_3
				}

				limit = {

# Monster Mobs for all monster provinces - Kekvit

	id = frmaint.21
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_save_game = no # First startup of the game
	}
	immediate = {
		any_title = {
			limit = {
				tier = COUNT
				location = {
					culture = monster
					holder = 0
				}
			}
			create_character = {
				name = "Monster Mob"
				dynasty = none
				culture = monster
				race = monster
			}
			new_character = {
				grant_title = PREV
			}
		}
	}
}
