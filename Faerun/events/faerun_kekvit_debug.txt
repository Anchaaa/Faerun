#
# Debug Events
#

namespace = FaerunDebug


#
# Silly Mode Toggle
#

character_event = {
	id = FaerunDebug.1
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = {
				has_global_flag = faerun_silly_mode
			}
			clr_global_flag = faerun_silly_mode
		}
		else = {
			set_global_flag = faerun_silly_mode
		}
	}
}
