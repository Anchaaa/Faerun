namespace = monstermain

character_event = {
	id = monstermain.1
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			AND = {
				culture = monster
				NOT = { religion = monster_pantheon }
			}
			AND = {
				NOT = { culture = monster }
				religion = monster_pantheon
			}
		}
	}
	
	immediate = {
		if = {
			limit = {
				culture = monster
				NOT = { religion = monster_pantheon }
			}
			religion = monster_pantheon
			break = yes
		}
		if = {
			limit = {
				NOT = { culture = monster }
				religion = monster_pantheon
			}
			religion = FROM
			break = yes
		}
	}
}
