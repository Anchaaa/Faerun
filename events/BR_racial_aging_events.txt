namespace = racial_aging

#This first is hidden and triggers for the long lived races and then sends them to racial_aging.1 where they become infirm at a certain age based on their health
character_event = {
        id = racial_aging.0
		min_age = 160
		hide_window = yes
		
		is_triggered_only = yes #from on_decade pulse
	
        trigger = {
                NOT = { has_character_flag = racial_aging0 }
				BR_long_lived_races = yes
				OR = {
				AND = {
				trait = creature_elf
				NOT = { trait = immortal }
				age = 350
				}
				AND = {
				trait = creature_feyri
				NOT = { trait = immortal }
				age = 350
				}
				AND = {
				trait = creature_drow
				NOT = { trait = immortal }
				age = 350
				}
				AND = {
				trait = creature_dwarf
				NOT = { trait = immortal }
				age = 180
				}
				AND = {
				trait = creature_half_elf
				NOT = { trait = immortal }
				age = 160
				}
				AND = {
				trait = longevity
				age = 200
				}
				}
				}
				
			immediate = {
		set_character_flag = racial_aging0
		}
		
		option = {
		if = {
			limit = {
			trait = creature_elf
			health = 9
			}
			character_event = { id = racial_aging.1 years = 500 }
			}	
		if = {
			limit = {
			trait = creature_elf
			health = 5
			}
			character_event = { id = racial_aging.1 years = 150 }
			}	
		if = {
			limit = {
			trait = creature_elf
			health = 1
			}
			character_event = { id = racial_aging.1 years = 25 }
			}	
		if = {
			limit = {
			trait = creature_feyri
			health = 9
			}
			character_event = { id = racial_aging.1 years = 500 }
			}	
		if = {
			limit = {
			trait = creature_feyri
			health = 5
			}
			character_event = { id = racial_aging.1 years = 150 }
			}	
		if = {
			limit = {
			trait = creature_feyri
			health = 1
			}
			character_event = { id = racial_aging.1 years = 25 }
			}	
		if = {
			limit = {
			trait = creature_drow
			health = 9
			}
			character_event = { id = racial_aging.1 years = 500 }
			}	
		if = {
			limit = {
			trait = creature_drow
			health = 5
			}
			character_event = { id = racial_aging.1 years = 150 }
			}	
		if = {
			limit = {
			trait = creature_drow
			health = 1
			}
			character_event = { id = racial_aging.1 years = 25 }
			}		
		if = {
			limit = {
			trait = creature_dwarf
			health = 9
			}
			character_event = { id = racial_aging.1 years = 200 }
			}	
		if = {
			limit = {
			trait = creature_dwarf
			health = 5
			}
			character_event = { id = racial_aging.1 years = 75 }
			}	
		if = {
			limit = {
			trait = creature_dwarf
			health = 1
			}
			character_event = { id = racial_aging.1 years = 25 }
