# Supported types: character

focus_wizardry = {
	type = character
	
	modifier_name = "focus_wizardry"
	
	potential = {
		is_playable = yes
		z_wizard = ye
	}
	
	allow = {
		is_adult = yes
		NOT = { trait = incapable }
		prisoner = no
		z_wizard = yes
	}
	
	# AI Pick chance - based on personality, not needs!
	chance = {
		factor = 100
	}
	
	# Focuses have no success conditions
	success = {
		always = no
	}
	
	abort = {
		OR = {
			prisoner = yes
			trait = incapable
			is_playable = no
		}
	}
	
	abort_effect = {
	}
	
	# MODIFIERS
}
